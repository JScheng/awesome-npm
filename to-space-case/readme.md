### to-space-case

#### useage

```javascript
   var toSpaceCase = require('to-space-case'); 

    toSpaceCase('camelCase')             // "camel case"
    toSpaceCase('snake_case')            // "snake case"
    toSpaceCase('dot.case')              // "dot case"
    toSpaceCase('-RAnDom -jUNk$__loL!')  // "random junk lol"

```
